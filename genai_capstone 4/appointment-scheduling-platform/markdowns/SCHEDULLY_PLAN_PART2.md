## 5. Stage 0 — Corpus Engineering

**File:** `backend/schedully/corpus_engineer.py`

Key changes from original plan:
- Added `.xlsx` parsing using `openpyxl` — this is the primary format for all report exports (appointments, users, providers, provider schedule, customer history) from the Node.js report-service
- `certificates_urls` in `provider_onboarding` is a comma-separated string of paths like `/uploads/provider_onboarding/<uuid>.pdf` — the auto-ingest endpoint resolves these to absolute disk paths under `backend/uploads/provider_onboarding/`
- Identity proof (`identity_proof_url`) and profile photo (`profile_photo_url`) are explicitly excluded from auto-ingest

```python
# backend/schedully/corpus_engineer.py

import os
import json
import hashlib
import re
import time
from pathlib import Path
from typing import Optional

import faiss
import numpy as np
from sentence_transformers import SentenceTransformer
from langchain_text_splitters import RecursiveCharacterTextSplitter

try:
    from pypdf import PdfReader
    PYPDF_AVAILABLE = True
except ImportError:
    PYPDF_AVAILABLE = False

try:
    from docx import Document as DocxDocument
    DOCX_AVAILABLE = True
except ImportError:
    DOCX_AVAILABLE = False

try:
    import openpyxl
    XLSX_AVAILABLE = True
except ImportError:
    XLSX_AVAILABLE = False

BASE_DIR = Path(__file__).resolve().parent
INDEX_DIR = BASE_DIR / "index"
INDEX_DIR.mkdir(parents=True, exist_ok=True)

FAISS_INDEX_PATH = str(INDEX_DIR / "faiss.index")
CHUNKS_JSON_PATH = str(INDEX_DIR / "chunks.json")

EMBEDDING_MODEL = "all-MiniLM-L6-v2"
EMBEDDING_DIM   = 384
CHUNK_SIZE      = 2000
CHUNK_OVERLAP   = 250

# Path to where provider onboarding files are stored on disk
# Matches UPLOAD_DIR in backend/routers/auth.py
PROVIDER_ONBOARDING_UPLOAD_DIR = Path(__file__).resolve().parents[1] / "uploads" / "provider_onboarding"

_embed_model: Optional[SentenceTransformer] = None

def get_embed_model() -> SentenceTransformer:
    global _embed_model
    if _embed_model is None:
        _embed_model = SentenceTransformer(EMBEDDING_MODEL)
    return _embed_model


def load_or_create_index() -> tuple[faiss.Index, list[dict]]:
    if os.path.exists(FAISS_INDEX_PATH) and os.path.exists(CHUNKS_JSON_PATH):
        index = faiss.read_index(FAISS_INDEX_PATH)
        with open(CHUNKS_JSON_PATH, "r") as f:
            chunks = json.load(f)
    else:
        index = faiss.IndexFlatL2(EMBEDDING_DIM)
        chunks = []
    return index, chunks


def save_index(index: faiss.Index, chunks: list[dict]) -> None:
    faiss.write_index(index, FAISS_INDEX_PATH)
    with open(CHUNKS_JSON_PATH, "w") as f:
        json.dump(chunks, f)


def _extract_pdf(path: str) -> str:
    if not PYPDF_AVAILABLE:
        raise ImportError("pypdf not installed. Run: pip install pypdf")
    reader = PdfReader(path)
    parts = []
    for page in reader.pages:
        text = page.extract_text()
        if text:
            parts.append(text)
    if not parts:
        raise ValueError(
            f"PDF '{path}' appears to be scanned (no extractable text). "
            "Use an OCR tool (e.g. ocrmypdf) to add a text layer first."
        )
    return "\n".join(parts)


def _extract_docx(path: str) -> str:
    if not DOCX_AVAILABLE:
        raise ImportError("python-docx not installed. Run: pip install python-docx")
    doc = DocxDocument(path)
    paragraphs = [p.text for p in doc.paragraphs if p.text.strip()]
    for table in doc.tables:
        for row in table.rows:
            row_texts = [cell.text.strip() for cell in row.cells if cell.text.strip()]
            if row_texts:
                paragraphs.append(" | ".join(row_texts))
    return "\n".join(paragraphs)


def _extract_xlsx(path: str) -> str:
    """
    Extract text from XLSX files generated by the Node.js report-service.

    Report sheets and their columns:
      All Appointments : appointment_id, customer_name, customer_email,
                         provider_name, specialization, category,
                         appointment_date, time_slot, status, is_paid,
                         fee_inr, notes, cancellation_reason, created_at
      Users            : id, full_name, email, phone, role, is_active, created_at
      Service Providers: id, provider_name, email, phone, specialization,
                         category, experience_years, location, avg_rating,
                         total_reviews, consultation_fee, is_verified,
                         is_accepting_appointments, created_at
      Schedule_M_Y     : appointment_date, time_slot, customer_name,
                         customer_email, customer_phone, status, is_paid,
                         fee_inr, notes, booked_at
      Appointment Hist : appointment_date, time_slot, provider_name,
                         specialization, category, status, is_paid,
                         fee_inr, notes, cancellation_reason, booked_at
    Each row is converted to a readable sentence so BM25 and the LLM
    can reason about it naturally.
    """
    if not XLSX_AVAILABLE:
        raise ImportError("openpyxl not installed. Run: pip install openpyxl")

    wb = openpyxl.load_workbook(path, read_only=True, data_only=True)
    parts = []

    for sheet_name in wb.sheetnames:
        ws = wb[sheet_name]
        parts.append(f"=== Sheet: {sheet_name} ===")

        rows_iter = ws.iter_rows(values_only=True)
        headers = None
        row_count = 0

        for row in rows_iter:
            if headers is None:
                # First row = headers
                headers = [str(h).strip() if h is not None else f"col_{i}"
                           for i, h in enumerate(row)]
                continue

            # Convert each data row to "key: value, key: value" sentence
            row_parts = []
            for h, val in zip(headers, row):
                if val is not None and str(val).strip():
                    row_parts.append(f"{h}: {val}")
            if row_parts:
                parts.append(", ".join(row_parts))
                row_count += 1

        parts.append(f"(Total rows in {sheet_name}: {row_count})")

    wb.close()
    return "\n".join(parts)


def extract_text(path: str) -> str:
    ext = Path(path).suffix.lower()
    if ext == ".pdf":
        return _extract_pdf(path)
    elif ext == ".docx":
        return _extract_docx(path)
    elif ext in (".xlsx", ".xls"):
        return _extract_xlsx(path)
    else:
        raise ValueError(f"Unsupported file type: '{ext}'. Supported: .pdf .docx .xlsx")


def canonicalize(text: str) -> str:
    text = re.sub(r"\r\n|\r", "\n", text)
    text = re.sub(r"\n{3,}", "\n\n", text)
    text = re.sub(r"[ \t]+", " ", text)
    text = re.sub(r"(?m)^\s*\d+\s*$", "", text)
    return text.strip()


def chunk_text(text: str, source_filename: str) -> list[dict]:
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=CHUNK_SIZE,
        chunk_overlap=CHUNK_OVERLAP,
        separators=["\n\n", "\n", " ", ""],
    )
    raw_chunks = splitter.split_text(text)
    return [
        {"source": source_filename, "chunk_index": i, "text": c}
        for i, c in enumerate(raw_chunks)
    ]


def fingerprint(text: str) -> str:
    normalized = re.sub(r"\s+", " ", text).strip().lower()
    return hashlib.sha256(normalized.encode()).hexdigest()


def ingest_file(
    file_path: str,
    tenant_id: str,
    original_filename: str,
) -> dict:
    """
    Full corpus engineering pipeline for one file.
    Supports: .pdf, .docx, .xlsx
    Returns: {filename, chunks_added, duplicates_skipped, doc_id}
    """
    file_size_mb = os.path.getsize(file_path) / (1024 * 1024)
    if file_size_mb > 20:
        raise RuntimeError(
            f"File '{original_filename}' is {file_size_mb:.1f} MB. Maximum is 20 MB."
        )

    raw_text = extract_text(file_path)
    text = canonicalize(raw_text)

    if len(text.strip()) < 50:
        raise ValueError(
            f"File '{original_filename}' produced less than 50 characters of text."
        )

    doc_id = fingerprint(text)
    index, existing_chunks = load_or_create_index()

    existing_doc_ids = {c.get("doc_id") for c in existing_chunks}
    if doc_id in existing_doc_ids:
        return {
            "filename": original_filename,
            "chunks_added": 0,
            "duplicates_skipped": 0,
            "doc_id": doc_id,
            "message": "Document already ingested (identical fingerprint). Skipped.",
        }

    chunks = chunk_text(text, original_filename)
    ingest_ts = int(time.time())

    existing_chunk_fps = {c.get("fingerprint") for c in existing_chunks}
    new_chunks = []
    duplicates_skipped = 0

    for c in chunks:
        fp = fingerprint(c["text"])
        if fp in existing_chunk_fps:
            duplicates_skipped += 1
            continue
        c["fingerprint"] = fp
        c["doc_id"] = doc_id
        c["tenant_id"] = tenant_id
        c["ingest_ts"] = ingest_ts
        new_chunks.append(c)

    if not new_chunks:
        return {
            "filename": original_filename,
            "chunks_added": 0,
            "duplicates_skipped": duplicates_skipped,
            "doc_id": doc_id,
            "message": "All chunks were duplicates. Nothing new added.",
        }

    model = get_embed_model()
    embeddings = model.encode(
        [c["text"] for c in new_chunks], show_progress_bar=False
    ).astype("float32")

    start_id = len(existing_chunks)
    for i, c in enumerate(new_chunks):
        c["faiss_id"] = start_id + i

    index.add(embeddings)
    save_index(index, existing_chunks + new_chunks)

    return {
        "filename": original_filename,
        "chunks_added": len(new_chunks),
        "duplicates_skipped": duplicates_skipped,
        "doc_id": doc_id,
    }


def ingest_provider_certificates(
    provider_id: str,
    tenant_id: str,
    certificates_urls: str,
) -> list[dict]:
    """
    Auto-ingest a provider's certificate/licence files.
    `certificates_urls` is the comma-separated string stored in
    provider_onboarding.certificates_urls, e.g.:
      "/uploads/provider_onboarding/abc123.pdf,/uploads/provider_onboarding/def456.pdf"

    Files are read from PROVIDER_ONBOARDING_UPLOAD_DIR on disk.
    Identity proofs and profile photos are NEVER passed here — only certificates_urls.

    Returns a list of ingest results (one per file).
    """
    results = []
    if not certificates_urls or not certificates_urls.strip():
        return results

    urls = [u.strip() for u in certificates_urls.split(",") if u.strip()]

    for url in urls:
        # url example: /uploads/provider_onboarding/abc123.pdf
        filename = Path(url).name
        absolute_path = PROVIDER_ONBOARDING_UPLOAD_DIR / filename

        if not absolute_path.exists():
            results.append({
                "filename": filename,
                "chunks_added": 0,
                "error": f"File not found on disk: {absolute_path}",
            })
            continue

        try:
            result = ingest_file(
                file_path=str(absolute_path),
                tenant_id=tenant_id,
                original_filename=f"certificate_{filename}",
            )
            results.append(result)
        except Exception as exc:
            results.append({
                "filename": filename,
                "chunks_added": 0,
                "error": str(exc),
            })

    return results


def list_documents(tenant_id: str) -> list[dict]:
    _, chunks = load_or_create_index()
    seen: dict[str, dict] = {}
    for c in chunks:
        if c.get("tenant_id") != tenant_id:
            continue
        did = c["doc_id"]
        if did not in seen:
            seen[did] = {
                "doc_id": did,
                "filename": c["source"],
                "ingest_ts": c.get("ingest_ts"),
                "chunk_count": 1,
            }
        else:
            seen[did]["chunk_count"] += 1
    return list(seen.values())


def delete_document(doc_id: str, tenant_id: str) -> int:
    _, chunks = load_or_create_index()
    kept_chunks = [
        c for c in chunks
        if not (c.get("doc_id") == doc_id and c.get("tenant_id") == tenant_id)
    ]
    removed = len(chunks) - len(kept_chunks)
    if removed == 0:
        return 0

    new_index = faiss.IndexFlatL2(EMBEDDING_DIM)
    if kept_chunks:
        model = get_embed_model()
        embeddings = model.encode(
            [c["text"] for c in kept_chunks], show_progress_bar=False
        ).astype("float32")
        new_index.add(embeddings)
        for i, c in enumerate(kept_chunks):
            c["faiss_id"] = i

    save_index(new_index, kept_chunks)
    return removed

## 6. Stage 1 — Understand + Plan

**File:** `backend/schedully/planner.py`

Key changes from original:
- Added `INTENT_REPORT` for questions about exported XLSX/PDF reports or certificates
- `INTENT_DOCUMENT` is now specifically for ad-hoc uploaded docs
- Customer role never gets `INTENT_LIVE` (no live tool access for customers)

```python
# backend/schedully/planner.py

import json
import os
import re
from dataclasses import dataclass, field
from typing import Optional
import openai

OPENROUTER_API_KEY = os.environ.get("OPENROUTER_API_KEY", "")
OPENROUTER_BASE_URL = "https://openrouter.ai/api/v1"
PLANNER_MODEL = "z-ai/glm-4.7-flash"

INTENT_KB       = "knowledge_base"      # "how do I…", product help
INTENT_LIVE     = "live_data"           # "show me my appointments today"
INTENT_DOCUMENT = "document_qa"         # uploaded doc / ingested certificate / report Q&A
INTENT_COMPOUND = "compound"            # needs both doc and live data
INTENT_CLARIFY  = "needs_clarification"
INTENT_OOS      = "out_of_scope"


@dataclass
class Plan:
    intent: str
    sub_queries: list[str] = field(default_factory=list)
    clarification_question: Optional[str] = None
    tool_hints: list[str] = field(default_factory=list)
    raw_query: str = ""


PLANNER_SYSTEM = """You are the intent planner for Schedully, an AI assistant embedded in Schedex
(an appointment scheduling platform).

Classify the user message into exactly one intent:
- "knowledge_base"       : product how-to, feature explanation, navigation help, policy questions
- "live_data"            : needs real-time appointment / provider / slot data from the platform API
- "document_qa"          : question about an ingested document — exported XLSX report, PDF certificate, uploaded file
- "compound"             : needs BOTH ingested document content AND live API data
- "needs_clarification"  : ambiguous, cannot be answered without more info
- "out_of_scope"         : nothing to do with Schedex

Also produce:
- sub_queries: 1–3 focused retrieval queries (split compound questions)
- clarification_question: if needs_clarification, one focused question
- tool_hints: API tools needed for live_data (from: get_appointments, get_providers,
  get_slots, get_dashboard, get_availability, get_reviews)

Reply ONLY with valid JSON:
{
  "intent": "<intent>",
  "sub_queries": ["<query1>", ...],
  "clarification_question": "<question or null>",
  "tool_hints": ["<tool1>", ...]
}"""


def classify_and_plan(
    user_message: str,
    conversation_context: str = "",
    user_role: str = "customer",
    max_retries: int = 2,
) -> Plan:
    """
    Stage 1: classify intent and produce sub-queries.
    For customer role, live_data and compound intents are downgraded to knowledge_base
    since customers have no live tool access in Schedully.
    """
    messages = []
    if conversation_context:
        messages.append({
            "role": "user",
            "content": f"[Conversation so far]\n{conversation_context}\n\n[New message]\n{user_message}",
        })
    else:
        messages.append({"role": "user", "content": user_message})

    for attempt in range(max_retries + 1):
        try:
            client = openai.OpenAI(
                base_url=OPENROUTER_BASE_URL,
                api_key=OPENROUTER_API_KEY,
            )
            response = client.chat.completions.create(
                model=PLANNER_MODEL,
                messages=[
                    {"role": "system", "content": PLANNER_SYSTEM},
                    *messages,
                ],
                max_tokens=300,
                temperature=0.0,
            )
            raw = response.choices[0].message.content or "{}"
            raw = re.sub(r"```(?:json)?", "", raw).strip()
            parsed = json.loads(raw)

            intent = parsed.get("intent", INTENT_KB)

            # Customers cannot use live_data tools in Schedully
            if user_role == "customer" and intent in (INTENT_LIVE, INTENT_COMPOUND):
                intent = INTENT_KB

            return Plan(
                intent=intent,
                sub_queries=parsed.get("sub_queries", [user_message]),
                clarification_question=parsed.get("clarification_question"),
                tool_hints=parsed.get("tool_hints", []),
                raw_query=user_message,
            )
        except (json.JSONDecodeError, KeyError, openai.OpenAIError):
            if attempt == max_retries:
                return Plan(
                    intent=INTENT_KB,
                    sub_queries=[user_message],
                    raw_query=user_message,
                )

    return Plan(intent=INTENT_KB, sub_queries=[user_message], raw_query=user_message)


def is_prompt_injection(text: str) -> bool:
    patterns = [
        r"ignore\s+(previous|above|all|prior)\s+instructions?",
        r"disregard\s+(previous|above|all|prior)\s+instructions?",
        r"you\s+are\s+now\s+(a|an)\s+",
        r"jailbreak",
        r"DAN\s+mode",
        r"pretend\s+you\s+(are|have\s+no)",
        r"reveal\s+(your\s+)?(system\s+)?prompt",
        r"print\s+(your\s+)?(system\s+)?prompt",
        r"show\s+(me\s+)?your\s+instructions",
    ]
    combined = re.compile("|".join(patterns), re.IGNORECASE)
    return bool(combined.search(text))

## 7. Stage 2 — Orchestrator

**File:** `backend/schedully/orchestrator.py`

No structural changes — the orchestrator is role-agnostic; role-based restrictions are enforced in the planner (customer role) and tool layer (route selection).

```python
# backend/schedully/orchestrator.py

import asyncio
from typing import Optional

from schedully.planner import (
    Plan, classify_and_plan, is_prompt_injection,
    INTENT_CLARIFY, INTENT_OOS, INTENT_LIVE, INTENT_KB,
    INTENT_DOCUMENT, INTENT_COMPOUND,
)
from schedully.hybrid_retriever import hybrid_retrieve, RetrievalResult
from schedully.tool_layer import run_tools
from schedully.reranker import fuse_and_rerank
from schedully.verifier import verify, VerificationResult
from schedully.generator import generate
from schedully.memory import SessionMemory

MAX_REPAIR_ATTEMPTS = 2


async def chat(
    user_message: str,
    session_memory: SessionMemory,
    tenant_id: str,
    user_id: str,
    user_role: str,
    bearer_token: str,
) -> dict:
    """
    Top-level orchestrator.
    Returns:
        {
            "answer": str,
            "sources": list[dict],
            "intent": str,
            "clarification": str | None,
            "repair_attempts": int,
            "verification": str,   # "grounded" | "weak"
        }
    """
    if is_prompt_injection(user_message):
        return {
            "answer": "I can't help with that. Is there something about Schedex I can assist you with?",
            "sources": [],
            "intent": "blocked",
            "clarification": None,
            "repair_attempts": 0,
            "verification": "grounded",
        }

    context_str = session_memory.get_context_string()

    plan: Plan = classify_and_plan(
        user_message, context_str, user_role=user_role
    )

    if plan.intent == INTENT_CLARIFY:
        session_memory.add_turn("user", user_message)
        session_memory.add_turn("assistant", plan.clarification_question or "Could you clarify?")
        return {
            "answer": plan.clarification_question or "Could you clarify your question?",
            "sources": [],
            "intent": INTENT_CLARIFY,
            "clarification": plan.clarification_question,
            "repair_attempts": 0,
            "verification": "grounded",
        }

    if plan.intent == INTENT_OOS:
        oos_reply = "I can only help with Schedex. Want me to look up your appointments or walk you through a feature?"
        session_memory.add_turn("user", user_message)
        session_memory.add_turn("assistant", oos_reply)
        return {
            "answer": oos_reply,
            "sources": [],
            "intent": INTENT_OOS,
            "clarification": None,
            "repair_attempts": 0,
            "verification": "grounded",
        }

    repair_attempt = 0
    top_k = 8

    while True:
        retrieval_task = None
        tool_task = None

        if plan.intent in (INTENT_KB, INTENT_DOCUMENT, INTENT_COMPOUND):
            retrieval_task = asyncio.create_task(
                asyncio.to_thread(
                    hybrid_retrieve,
                    sub_queries=plan.sub_queries,
                    tenant_id=tenant_id,
                    top_k=top_k,
                )
            )

        if plan.intent in (INTENT_LIVE, INTENT_COMPOUND):
            tool_task = asyncio.create_task(
                run_tools(
                    plan=plan,
                    tenant_id=tenant_id,
                    user_id=user_id,
                    user_role=user_role,
                    bearer_token=bearer_token,
                )
            )

        retrieved_chunks: list[RetrievalResult] = []
        tool_results: list[dict] = []

        if retrieval_task:
            retrieved_chunks = await retrieval_task
        if tool_task:
            tool_results = await tool_task

        fused = fuse_and_rerank(retrieved_chunks, plan.sub_queries)
        all_context = fused + tool_results
        verification: VerificationResult = verify(
            query=user_message,
            context_items=all_context,
            plan=plan,
        )

        if verification.status == "grounded" or repair_attempt >= MAX_REPAIR_ATTEMPTS:
            break

        repair_attempt += 1
        top_k = top_k + 4

    answer, sources = generate(
        query=user_message,
        context_items=all_context,
        plan=plan,
        verification=verification,
        conversation_context=context_str,
    )

    session_memory.add_turn("user", user_message)
    session_memory.add_turn("assistant", answer)

    return {
        "answer": answer,
        "sources": sources,
        "intent": plan.intent,
        "clarification": None,
        "repair_attempts": repair_attempt,
        "verification": verification.status,
    }
```
